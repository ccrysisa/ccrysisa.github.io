  • (e) 72

  • (b) 86
  • (e) 72
  • (a) 63
  • (c) 51
  • (d) 48

    for (list *merge = NULL; left || right; ) {
        if (!right || (left && left->data < right->data)) {
            if (!merge) {
                start = merge = left; // LL1;
            } else {
                merge->next = left; // LL2;
                merge = merge->next;
            }
            left = left->next; // LL3;
        } else {
            if (!merge) {
                start = merge = right; // LL4;

                merge = merge->next;
            }
